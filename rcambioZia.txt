export interface EnrollmentCapacityState {
  // Catálogos
  careers: TeacherDistributionCatalogue[];
  teacherDistributions: TeacherDistributionCatalogue[];
  levels: TeacherDistributionCatalogue[];

  // Datos crudos de la DB
  capacitiesRaw: CapacityRawInterface[];

  // Selecciones de la UI
  selectedCareerId: string | null;
  selectedTeacherDistributionId: string | null;
  selectedLevelId: string | null;

  // Estado del Modal
  showModal: boolean;
  modalCell: CellInterface | null;
  modalCapacityInput: number;
}

export interface TeacherDistributionCatalogue {
  id: string;
  name: string;
}

export interface CapacityRawInterface {
  id: string;
  teacherDistribution?: { id: string; name: string };
  career?: { id: string; name: string };
  academicPeriod?: { id: string; name: string };
  parallel?: { id: string; name: string };
  workday?: { id: string; name: string };
  capacity: number;
  studentsCount?: number;
}

// Interfaces para la matriz visual (antes en el componente)
export interface CellInterface {
  id: string;
  horario: string;
  paralelo: string;
  cupoMaximo: number;
  estudiantesContados: number;
  colorSemaforo: 'verde' | 'naranja' | 'rojo';
}

export interface BlockInterface {
  horarioNombre: string;
  celdas: CellInterface[];
}

export interface RowInterface {
  jornadaId: string;
  jornadaNombre: string;
  bloquesHorarios: BlockInterface[];
}

export const INITIAL_ENROLLMENT_CAPACITY_STATE: EnrollmentCapacityState = {
  careers: [],
  teacherDistributions: [],
  levels: [],
  capacitiesRaw: [],
  selectedCareerId: null,
  selectedTeacherDistributionId: null,
  selectedLevelId: null,
  showModal: false,
  modalCell: null,
  modalCapacityInput: 0,
};

import { computed, inject, Injectable, signal } from '@angular/core';
import {
  EnrollmentCapacityState,
  INITIAL_ENROLLMENT_CAPACITY_STATE,
  RowInterface,
  CapacityRawInterface,
  CellInterface
} from './enrollment-capacity.state';
import { EnrollmentCapacityService } from './enrollment-capacity.service';

const FORM_STATE_KEY = 'enrollment-capacity-v2';

@Injectable({ providedIn: 'root' })
export class EnrollmentCapacityStore {
  private readonly service = inject(EnrollmentCapacityService);

  readonly state = signal<EnrollmentCapacityState>(this.loadFromStorage());

  // --- DICCIONARIOS DE ORDENAMIENTO ESTÁTICO ---
  private readonly PRIORIDAD_NIVELES: Record<string, number> = {
    'Nivel A1.2': 1, 'Nivel A2.2': 2, 'Nivel B1.2': 3, 'Nivel B2.2': 4
  };
  private readonly PRIORIDAD_JORNADAS: Record<string, number> = {
    'Matutina': 1, 'Vespertina': 2, 'Nocturna': 3
  };

  // --- COMPUTEDS DE CATÁLOGOS ---
  readonly careers = computed(() => this.state().careers);
  readonly teacherDistributions = computed(() => this.state().teacherDistributions);
  
  readonly sortedLevels = computed(() => {
    const lista = this.state().levels;
    return [...lista].sort((a, b) => (this.PRIORIDAD_NIVELES[a.name] || 99) - (this.PRIORIDAD_NIVELES[b.name] || 99));
  });

  readonly activeLevelName = computed(() => {
    const activo = this.sortedLevels().find(n => n.id === this.state().selectedLevelId);
    return activo ? activo.name : 'Seleccione un Nivel';
  });

  // --- COMPUTED DE MATRIZ (Lógica pura extraída del componente antiguo) ---
  readonly matrixRows = computed<RowInterface[]>(() => {
    const { capacitiesRaw, selectedCareerId, selectedTeacherDistributionId, selectedLevelId } = this.state();
    
    // 1. Filtrar datos crudos
    const filtrados = capacitiesRaw.filter(item =>
      item.career?.id === selectedCareerId &&
      item.teacherDistribution?.id === selectedTeacherDistributionId &&
      item.academicPeriod?.id === selectedLevelId
    );

    // 2. Agrupar por Jornadas (Ordenadas)
    const jornadasMap = new Map<string, { id: string; name: string }>();
    filtrados.forEach(item => {
      if (item.workday) jornadasMap.set(item.workday.id, item.workday);
    });

    const jornadasOrdenadas = Array.from(jornadasMap.values()).sort((a, b) => {
      const pA = this.PRIORIDAD_JORNADAS[a.name.split(' ')[0]] || 99;
      const pB = this.PRIORIDAD_JORNADAS[b.name.split(' ')[0]] || 99;
      return pA === pB ? a.name.localeCompare(b.name) : pA - pB;
    });

    // 3. Construir filas y celdas
    return jornadasOrdenadas.map(jornada => {
      const registrosDeJornada = filtrados.filter(f => f.workday?.id === jornada.id);
      
      const celdas: CellInterface[] = registrosDeJornada.map(reg => {
        const max = reg.capacity || 0;
        const ocupados = reg.studentsCount || 0;
        const porcentaje = max > 0 ? (ocupados / max) * 100 : 0;
        
        let semaforo: 'verde' | 'naranja' | 'rojo' = 'verde';
        if (porcentaje >= 100) semaforo = 'rojo';
        else if (porcentaje >= 80) semaforo = 'naranja';

        return {
          id: reg.id,
          horario: reg.workday?.name || jornada.name,
          paralelo: reg.parallel?.name || 'Sin Nombre',
          cupoMaximo: max,
          estudiantesContados: ocupados,
          colorSemaforo: semaforo
        };
      }).sort((a, b) => a.paralelo.localeCompare(b.paralelo)); // Orden A, B, C...

      return {
        jornadaId: jornada.id,
        jornadaNombre: jornada.name,
        bloquesHorarios: [{ horarioNombre: jornada.name, celdas }]
      };
    });
  });

  // --- COMPUTEDS DE MÉTRICAS Y GRÁFICOS ---
  readonly totalCupos = computed(() => this.matrixRows().reduce((sum, f) => sum + f.bloquesHorarios[0].celdas.reduce((s, c) => s + c.cupoMaximo, 0), 0));
  readonly totalOcupados = computed(() => this.matrixRows().reduce((sum, f) => sum + f.bloquesHorarios[0].celdas.reduce((s, c) => s + c.estudiantesContados, 0), 0));
  readonly totalDisponibles = computed(() => Math.max(0, this.totalCupos() - this.totalOcupados()));
  readonly porcentajeOcupacion = computed(() => this.totalCupos() === 0 ? 0 : Math.round((this.totalOcupados() / this.totalCupos()) * 100));

  readonly chartData = computed(() => ({
    labels: ['Ocupado', 'Disponible'],
    datasets: [{ data: [this.totalOcupados(), this.totalDisponibles()], backgroundColor: ['#f97316', '#22c55e'], hoverBackgroundColor: ['#ea580c', '#16a34a'], borderWidth: 0 }]
  }));

  // --- ACCIONES (SETTERS) ---
  selectCareer(id: string | null) { this.state.update(s => ({ ...s, selectedCareerId: id, selectedLevelId: null })); }
  selectDistribution(id: string | null) { this.state.update(s => ({ ...s, selectedTeacherDistributionId: id, selectedLevelId: null })); }
  selectLevel(id: string) { this.state.update(s => ({ ...s, selectedLevelId: id })); }

  openModal(cell: CellInterface) {
    this.state.update(s => ({ ...s, showModal: true, modalCell: cell, modalCapacityInput: cell.cupoMaximo }));
  }
  closeModal() { this.state.update(s => ({ ...s, showModal: false, modalCell: null })); }
  setModalInput(val: number) { this.state.update(s => ({ ...s, modalCapacityInput: val })); }

  // --- ACCIONES HTTP (CRUD INMUTABLE) ---
  loadInitialData() {
    this.service.getCatalogues().subscribe({
      next: (data: any) => {
        this.state.update(s => ({
          ...s,
          careers: data.careers || [],
          teacherDistributions: data.teacherDistributions || [],
          levels: data.levels || [],
          selectedCareerId: data.careers?.[0]?.id || null,
          selectedTeacherDistributionId: data.teacherDistributions?.[0]?.id || null
        }));
        // Auto-seleccionar primer nivel
        const firstLevel = this.sortedLevels()[0]?.id;
        if (firstLevel) this.selectLevel(firstLevel);
      }
    });

    this.service.findAll().subscribe({
      next: (data: any) => this.state.update(s => ({ ...s, capacitiesRaw: data.data || data }))
    });
  }

  saveCapacity() {
    const { modalCell, modalCapacityInput } = this.state();
    if (!modalCell) return;

    // Optimistic UI Update
    const newRaw = this.state().capacitiesRaw.map(r => r.id === modalCell.id ? { ...r, capacity: modalCapacityInput } : r);
    this.state.update(s => ({ ...s, capacitiesRaw: newRaw, showModal: false }));

    this.service.update(modalCell.id, { capacity: modalCapacityInput }).subscribe();
  }

  deleteCapacity() {
    const { modalCell } = this.state();
    if (!modalCell || modalCell.estudiantesContados > 0) return;

    const newRaw = this.state().capacitiesRaw.filter(r => r.id !== modalCell.id);
    this.state.update(s => ({ ...s, capacitiesRaw: newRaw, showModal: false }));

    this.service.remove(modalCell.id).subscribe();
  }

  private loadFromStorage(): EnrollmentCapacityState {
    const stored = sessionStorage.getItem(FORM_STATE_KEY);
    return stored ? JSON.parse(stored) : INITIAL_ENROLLMENT_CAPACITY_STATE;
  }
}

import { inject, Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { environment } from '../../../../environments/environment';
import { Observable } from 'rxjs';
import { map } from 'rxjs/operators';
import { HttpResponseInterface } from '../../../auth/interfaces/http-response.interface';

@Injectable({
    providedIn: 'root'
})
export class EnrollmentCapacityHttpService {
    private readonly _httpClient = inject(HttpClient);
    private readonly _apiUrl = `${environment.API_URL}/admin/enrollment-capacity`;

    /**
     * Recupera los catálogos de carreras, distribuciones docentes y niveles académicos
     */
    findCatalogues(): Observable<any> {
        return this._httpClient.get<HttpResponseInterface>(`${this._apiUrl}/catalogues`).pipe(
            map((response) => response.data)
        );
    }

    /**
     * Envía la configuración final calculada de cupos de matrícula al servidor
     */
    register(payload: any): Observable<any> {
        return this._httpClient.post<HttpResponseInterface>(this._apiUrl, payload).pipe(
            map((response) => response.data)
        );
    }

    // --- NUEVOS MÉTODOS AÑADIDOS PARA QUE EL STORE PUEDA HACER CRUD EN EL MODAL ---
    
    /**
     * Obtiene todos los cupos existentes para pintar la matriz
     */
    findAll(): Observable<any> {
        return this._httpClient.get<HttpResponseInterface>(this._apiUrl).pipe(
            map((response) => response.data)
        );
    }

    /**
     * Actualiza el cupo máximo de un paralelo específico (Lo usa el modal al guardar)
     */
    update(id: string, payload: { capacity: number }): Observable<any> {
        return this._httpClient.patch<HttpResponseInterface>(`${this._apiUrl}/${id}`, payload).pipe(
            map((response) => response.data)
        );
    }

    /**
     * Elimina un cupo/paralelo (Lo usa el modal al borrar)
     */
    remove(id: string): Observable<any> {
        return this._httpClient.delete<HttpResponseInterface>(`${this._apiUrl}/${id}`).pipe(
            map((response) => response.data)
        );
    }
}
