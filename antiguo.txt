// --- INTERFACES DE CONTROL VISUAL ---
export interface CeldaHorarioInterface {
  id: string;
  horario: string;
  paralelo: string;
  cupoMaximo: number;
  estudiantesContados: number;
  colorSemaforo: 'verde' | 'naranja' | 'rojo';
}

export interface BloqueHorarioInterface {
  horarioNombre: string;
  celdas: CeldaHorarioInterface[];
}

export interface FilaJornadaInterface {
  jornadaId: string;
  jornadaNombre: string;
  bloquesHorarios: BloqueHorarioInterface[];
}

@Component({
  selector: 'app-enrollment-capacity-form',
  standalone: true,
  imports: [
    CommonModule,
    FormsModule,
    SelectModule,
    ButtonModule,
    DialogModule,
    InputTextModule,
    ChartModule
  ],
  templateUrl: './enrollment-capacity-form.html'
})
export class EnrollmentCapacityFormComponent implements OnInit {
  
  // --- DICCIONARIOS ESTÁTICOS DE ORDENAMIENTO INMUTABLE (CARPINTERÍA FINA) ---
  private readonly PRIORIDAD_NIVELES: Record<string, number> = {
    'Nivel A1.2': 1,
    'Nivel A2.2': 2,
    'Nivel B1.2': 3,
    'Nivel B2.2': 4
  };

  private readonly PRIORIDAD_JORNADAS: Record<string, number> = {
    'Matutina': 1,
    'Vespertina': 2,
    'Nocturna': 3
  };

  // --- SIGNALS DE ESTADO LOCAL ---
  readonly carreraSeleccionadaId = signal<string | null>(null);
  readonly periodoLectivoId = signal<string | null>(null);
  readonly nivelSeleccionadoId = signal<string | null>(null);

  // Mocks/Data general mapeada desde tu servicio
  readonly careers = signal<any[]>([]);
  readonly schoolPeriods = signal<any[]>([]);
  
  // Cache de datos crudos directo de la Base de Datos
  readonly todosLosNivelesRaw = signal<any[]>([]);
  readonly todosLosCuposRaw = signal<any[]>([]);

  // Control del Modal CRUD
  readonly mostrarModalCrud = signal<boolean>(false);
  readonly celdaSeleccionada = signal<CeldaHorarioInterface | null>(null);
  readonly cupoEdicionInput = signal<number>(0);

  // --- SIGNALS COMPUTADOS ORDENADOS Y ESTÁTICOS ---

  // 1. Niveles ordenados fijamente: A1.2 -> A2.2 -> B1.2 -> B2.2
  readonly academicPeriods = computed<any[]>(() => {
    const listaOriginal = this.todosLosNivelesRaw() || [];
    return [...listaOriginal].sort((a, b) => {
      const pesoA = this.PRIORIDAD_NIVELES[a.name] || 99;
      const pesoB = this.PRIORIDAD_NIVELES[b.name] || 99;
      return pesoA - pesoB;
    });
  });

  // Nombre del nivel activo para el título de la matriz
  readonly nombreNivelActivo = computed<string>(() => {
    const activo = this.academicPeriods().find(n => n.id === this.nivelSeleccionadoId());
    return activo ? activo.name : 'Seleccione un Nivel';
  });

  // 2. Jornadas únicas ordenadas fijamente: Matutinas -> Vespertinas -> Nocturnas
  readonly workdays = computed<any[]>(() => {
    const data = this.todosLosCuposRaw();
    const carreraId = this.carreraSeleccionadaId();
    const periodoId = this.periodoLectivoId();
    const nivelId = this.nivelSeleccionadoId();

    const jornadasMap = new Map();
    data.forEach(item => {
      if (item.schoolPeriod?.id === periodoId && item.career?.id === carreraId && item.academicPeriod?.id === nivelId && item.workday) {
        jornadasMap.set(item.workday.id, item.workday);
      }
    });

    return Array.from(jornadasMap.values()).sort((a, b) => {
      const nombreA = a.name.split(' ')[0];
      const nombreB = b.name.split(' ')[0];

      const pesoA = this.PRIORIDAD_JORNADAS[nombreA] || 99;
      const pesoB = this.PRIORIDAD_JORNADAS[nombreB] || 99;

      if (pesoA === pesoB) {
        return a.name.localeCompare(b.name); // Desempate alfabético interno por horas
      }
      return pesoA - pesoB;
    });
  });

  // 3. Matriz estructural limpia blindada contra desordenamiento del backend
  readonly matrizJornadas = computed<FilaJornadaInterface[]>(() => {
    const dataRaw = this.todosLosCuposRaw();
    const careerId = this.carreraSeleccionadaId();
    const periodId = this.periodoLectivoId();
    const levelId = this.nivelSeleccionadoId();
    const listaJornadasOrdered = this.workdays();

    const filtrados = dataRaw.filter(item => 
        item.career?.id === careerId &&
        item.schoolPeriod?.id === periodId &&
        item.academicPeriod?.id === levelId
    );

    return listaJornadasOrdered.map(jornada => {
        const registrosDeEstaJornada = filtrados.filter(f => f.workday?.id === jornada.id);
        
        const celdas: CeldaHorarioInterface[] = registrosDeEstaJornada.map(reg => {
            const max = reg.capacity || 0;
            const ocupados = reg.studentsCount || reg.enrolledStudents || 0;
            const porcentaje = max > 0 ? (ocupados / max) * 100 : 0;
            
            let semaforo: 'verde' | 'naranja' | 'rojo' = 'verde';
            if (porcentaje >= 100) semaforo = 'rojo';
            else if (porcentaje >= 80) semaforo = 'naranja';

            return {
                id: reg.id,
                horario: reg.workday?.name || jornada.name, 
                paralelo: reg.parallel?.name || 'Sin Nombre',
                cupoMaximo: max,
                estudiantesContados: ocupados,
                colorSemaforo: semaforo
            };
        });

        // Garantizar de forma estricta que los paralelos mantengan orden A, B, C, D al actualizar
        celdas.sort((a, b) => a.paralelo.localeCompare(b.paralelo));

        return {
            jornadaId: jornada.id,
            jornadaNombre: jornada.name,
            bloquesHorarios: [{
                horarioNombre: jornada.name,
                celdas: celdas
            }]
        };
    });
  });

  // --- COMPUTADOS PARA MÉTRICAS Y GRÁFICO (DOUGHNUT) ---
  readonly totalCuposNivel = computed(() => {
    return this.matrizJornadas().reduce((sum, fila) => 
      sum + fila.bloquesHorarios[0].celdas.reduce((cSum, c) => cSum + c.cupoMaximo, 0), 0
    );
  });

  readonly totalOcupadosNivel = computed(() => {
    return this.matrizJornadas().reduce((sum, fila) => 
      sum + fila.bloquesHorarios[0].celdas.reduce((cSum, c) => cSum + c.estudiantesContados, 0), 0
    );
  });

  readonly totalDisponiblesNivel = computed(() => {
    const res = this.totalCuposNivel() - this.totalOcupadosNivel();
    return res < 0 ? 0 : res;
  });

  readonly porcentajeOcupacionGlobal = computed(() => {
    const total = this.totalCuposNivel();
    if (total === 0) return 0;
    return Math.round((this.totalOcupadosNivel() / total) * 100);
  });

  readonly chartData = computed(() => {
    return {
      labels: ['Ocupado', 'Disponible'],
      datasets: [
        {
          data: [this.totalOcupadosNivel(), this.totalDisponiblesNivel()],
          backgroundColor: ['#f97316', '#22c55e'],
          hoverBackgroundColor: ['#ea580c', '#16a34a'],
          borderWidth: 0
        }
      ]
    };
  });

  readonly chartOptions = signal({
    cutout: '78%',
    plugins: {
      legend: { display: false },
      tooltip: { enabled: true }
    }
  });

  // --- LÓGICA DE INICIALIZACIÓN ---
  ngOnInit(): void {
    this.cargarDatosSimulados();
  }

    // Aquí mandas a llamar tus servicios reales HTTP de backend

    this.schoolPeriods.set([
      { id: 'p1', name: 'Período 2026-1' },
      { id: 'p2', name: 'Periodo 2027-1' }
    ]);
    this.todosLosNivelesRaw.set([
      { id: 'n2', name: 'Nivel A2.2' },
      { id: 'n4', name: 'Nivel B2.2' },
      { id: 'n1', name: 'Nivel A1.2' },
      { id: 'n3', name: 'Nivel B1.2' }
    ]);

    // Selección por defecto
    this.carreraSeleccionadaId.set('c1');
    this.periodoLectivoId.set('p1');
    
    // Al setear el primer nivel ordenado
    if (this.academicPeriods().length > 0) {
      this.nivelSeleccionadoId.set(this.academicPeriods()[0].id);
    }

    // Datos simulados de la base de datos (Ejemplo inicial para pruebas)
    this.simularCuposBaseDeDatos();

  seleccionarNivel(id: string): void {
    this.nivelSeleccionadoId.set(id);
  }

  // --- MOCK CRUD CONTROL (GUARDADO INMUTABLE) ---
  abrirEdicionCupo(celda: CeldaHorarioInterface): void {

    this.cupoEdicionInput.set(celda.cupoMaximo);

  }

  guardarCupoModificado(): void {
    const celda = this.celdaSeleccionada();
    if (celda) {
      // Modificamos el valor directamente en el array crudo simulación de backend
      const copiaRaw = [...this.todosLosCuposRaw()];
      const indice = copiaRaw.findIndex(item => item.id === celda.id);
      
      if (indice !== -1) {
        copiaRaw[indice].capacity = this.cupoEdicionInput();
        this.todosLosCuposRaw.set(copiaRaw); // Se dispara el computed manteniendo el .sort estricto
      }
    }
    this.mostrarModalCrud.set(false);
  }

  eliminarCupoExistente(): void {
    const celda = this.celdaSeleccionada();
    if (celda && celda.estudiantesContados === 0) {
      const filtrados = this.todosLosCuposRaw().filter(item => item.id !== celda.id);
      this.todosLosCuposRaw.set(filtrados);
    }
    this.mostrarModalCrud.set(false);
  }

  private simularCuposBaseDeDatos(): void {

import {Component, inject, OnInit, signal} from '@angular/core';

import {TableModule} from 'primeng/table';

import {TooltipModule} from 'primeng/tooltip';

import {EnrollmentCapacityHttpService} from '../../services/enrollment-capacity-http.service';

@Component({
    selector: 'app-enrollment-capacity-list',

    standalone: true,

    imports: [
        TableModule,
        Button,
        ToastModule,
        TooltipModule,
    ],

    providers: [
        MessageService,
    ],

    templateUrl: './enrollment-capacity-list.html',
})
export class EnrollmentCapacityList implements OnInit {

    private readonly service =
        inject(EnrollmentCapacityHttpService);

    private readonly router =
        inject(Router);

    private readonly messageService =
        inject(MessageService);

    protected data =
        signal<any[]>([]);

    protected loading =
        signal(false);

    ngOnInit(): void {
        this.loadData();
    }

    loadData(): void {

        this.loading.set(true);

        this.service.findAll().subscribe({

            next: (response: any) => {

                this.data.set(
                    response.data ?? response
                );

                this.loading.set(false);

            },

            error: () => {

                this.loading.set(false);

                this.messageService.add({

                    severity: 'error',

                    summary: 'Error',

                    detail: 'No fue posible cargar los registros.',

                    life: 4000,

                });

            },

        });

    }

   create(): void {
    this.router.navigate([
        '/main/admin/enrollment-capacity/form'
    ]);
}

  edit(id: string): void {
    this.router.navigate([
        '/main/admin/enrollment-capacity/form',
        id,
    ]);
}

 remove(item: any): void {

    if (!confirm('¿Desea eliminar este registro?')) {
        return;
    }

    this.service.remove(item.id).subscribe({

        next: () => {

            this.messageService.add({

                severity: 'success',

                summary: 'Correcto',

                detail: 'Registro eliminado.',

            });

            this.loadData();

        },

        error: (err) => {

            this.messageService.add({

                severity: 'warn',

                summary: 'No permitido',

                detail:
                    err.error.message ??
                    'Existen estudiantes matriculados.',

            });

        }

    });

}
}


import { inject, Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { environment } from '@env/environment';
import { Observable } from 'rxjs';

@Injectable({
  providedIn: 'root'
})
export class EnrollmentCapacityHttpService {
  private readonly httpClient = inject(HttpClient);

  // Ruta base oficial unificada para interactuar con la tabla/entidad career-parallels
  private readonly apiUrl = `${environment.API_URL}/career-parallels`;

  /**
   * Recupera la lista completa de configuraciones, ofertas y catálogos vinculados
   */
  findAll(): Observable<any> {
    return this.httpClient.get<any>(this.apiUrl);
  }

  /**
   * Obtiene el detalle de un registro específico de capacidad por su ID
   */
  findOne(id: string): Observable<any> {
    return this.httpClient.get<any>(`${this.apiUrl}/${id}`);
  }

  /**
   * Envía los datos mapeados para insertar o actualizar la capacidad en la base de datos
   */
  create(payload: any): Observable<any> {
    return this.httpClient.post<any>(this.apiUrl, payload);
  }

  /**
   * Actualiza parcialmente un registro existente por su ID único
   */
  update(id: string, payload: any): Observable<any> {
    return this.httpClient.patch<any>(`${this.apiUrl}/${id}`, payload);
  }

  /**
   * Elimina una configuración de capacidad/paralelo de la base de datos
   */
  remove(id: string): Observable<any> {
    return this.httpClient.delete<any>(`${this.apiUrl}/${id}`);
  }

  /**
   * Consulta la lista o conteo de estudiantes inscritos en este paralelo/capacidad
   */
  countStudents(id: string): Observable<any> {
    return this.httpClient.get<any>(`${this.apiUrl}/${id}/students`);
  }
}

    configuration: {

        academicPeriodId: null,

        careerId: null,

        parallelId: null,

        workdayId: null,

        schoolPeriodId: null,

        capacity: null,

    }

};


import {computed, Injectable, signal} from '@angular/core';

import {

    EnrollmentCapacityInfo,
    EnrollmentCapacityState,
    INITIAL_ENROLLMENT_CAPACITY_STATE

} from './enrollment-capacity.state';

const FORM_STATE_KEY = 'enrollment-capacity';

@Injectable({
    providedIn: 'root'
})
export class EnrollmentCapacityStore {

    readonly formState = signal<EnrollmentCapacityState>(
        this.loadFromStorage()
    );

    readonly formErrors =
        signal<Record<string, string[]>>({});

    readonly configuration =
        computed(() => this.formState().configuration);

    updateConfiguration(
        data: Partial<EnrollmentCapacityInfo>
    ) {

        this.formState.update(state => ({

            ...state,

            configuration: {

                ...state.configuration,

                ...data,

            }

        }));

    }

    private loadFromStorage(): EnrollmentCapacityState {

        const stored =
            sessionStorage.getItem(FORM_STATE_KEY);

        return stored
            ? JSON.parse(stored)
            : INITIAL_ENROLLMENT_CAPACITY_STATE;

    }

}
